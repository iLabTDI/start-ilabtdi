# ─────────────────────────────────────────────────────────────
# iLab TDI · credenciales del hosting / DB
#
# Copia este archivo a `.credentials.txt` (no se commitea — está en .gitignore).
# Luego corre:   pnpm bootstrap
# Con eso se genera .env, .env.production.local, backend/config.php, y
# se suben los secrets al repo de GitHub (si tienes `gh` autenticado).
# ─────────────────────────────────────────────────────────────

# ── Hosting GoDaddy (FTP/SFTP) ──────────────────────────────
FTP_HOST=ftp.tudominio.com
FTP_USER=deploy@tudominio.com
FTP_PASS=
FTP_REMOTE_DIR=/public_html/
FTP_PROTOCOL=ftps    # ftps | ftp | sftp

# ── SSH (opcional, útil si usas SFTP o tunneling a MySQL) ──
SSH_HOST=
SSH_USER=
SSH_PORT=22
SSH_KEY_PATH=~/.ssh/id_ed25519   # ruta local a la llave privada

# ── MySQL en cPanel ─────────────────────────────────────────
# Si activas "Remote MySQL" en cPanel, pon la IP/host que GoDaddy te dio.
# Si no, vas a conectar via SSH tunnel (pon el host como `localhost`).
DB_HOST=localhost
DB_PORT=3306
DB_NAME=ilabtdi_app
DB_USER=ilabtdi_app
DB_PASS=

# ── Aplicación ──────────────────────────────────────────────
APP_URL=https://ilabtdi.com
APP_NAME=iLab TDI

# ── Auth backend ────────────────────────────────────────────
# "supabase" (default) o "php" (MySQL del hosting)
AUTH_BACKEND=supabase

# ── Supabase (si AUTH_BACKEND=supabase) ─────────────────────
SUPABASE_URL=
SUPABASE_ANON_KEY=
SUPABASE_PROJECT_ID=

# ── JWT (si AUTH_BACKEND=php) ───────────────────────────────
# Si dejas vacío, el bootstrap genera un secreto de 64 chars.
JWT_SECRET=

# ── GitHub (opcional, para auto-subir secrets al repo) ──────
# Si tienes `gh` autenticado, el bootstrap configura los secrets.
# Si prefieres hacerlo manual, deja GH_AUTO_PUSH=false.
GH_AUTO_PUSH=true
GH_REPO=ilabtdi/start-ilabtdi
